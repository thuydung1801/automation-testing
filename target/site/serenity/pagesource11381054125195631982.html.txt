    function CheckSearchString() {

        { return true; }

        { alert('Please enter search text !'); $('.searchtexbox_header').focus(); return false; }
    }

    $(document).ready(function() {

        $('.searchtexbox_header').bind("focusin", function(event) {
            event.stopPropagation();
            if ($(this).val() == 'Enter Search Text. . .') {
                $('.searchtexbox_header').val("");
            }
        });

        $('.searchtexbox_header').bind("focusout", function(event) {
            //event.stopPropagation();
            if ($(this).val() == "") {
                $('.searchtexbox_header').val("Enter Search Text. . .");
                $('.searchtexbox_header').animate({ width: "0px" });

                $('.main_searchbox').hide();
                $('.main_searchbox').css("border", "none");
                $('.search_icon_header').show();
            }
        });



        $('.search_icon_header').click(function(event) {

            event.stopPropagation();
            $('.searchtexbox_header').animate({ width: "140px" });
            $('.main_searchbox').show();
            $('.main_searchbox').css("border", "1px solid black");
            $('.search_icon_header').hide();
        });

        $('html').click(function(e) {
            var $target = $(e.target);
            if ($target.closest('#ctl00_CPHHeader_he1_txtSearch').length == 0) {
                if ($('.searchtexbox_header').val() == "") {
                    $('.searchtexbox_header').val("Enter Search Text. . .");
                }
                $('.searchtexbox_header').animate({ width: "0px" });

                $('.main_searchbox').hide();
                $('.main_searchbox').css("border", "none");
                $('.search_icon_header').show();
            }


        });




    });
